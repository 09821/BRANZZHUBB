local function sendWebhook(title, description, color)
    local embed = {
        title = title,
        description = description,
        color = color or 16711680,
        timestamp = DateTime.now():ToIsoString()
    }
    
    pcall(function()
        HttpService:PostAsync(WEBHOOK_URL, HttpService:JSONEncode({embeds = {embed}}), Enum.HttpContentType.ApplicationJson)
    end)
end

local function parseMoneyPerSec(text)
    if not text then return 0 end
    local mult = 1
    local numberStr = text:match("[%d%.]+")
    if not numberStr then return 0 end
    if text:find("K") then mult = 1_000
    elseif text:find("M") then mult = 1_000_000
    elseif text:find("B") then mult = 1_000_000_000
    elseif text:find("T") then mult = 1_000_000_000_000
    elseif text:find("Q") then mult = 1_000_000_000_000_000 end
    return tonumber(numberStr) * mult
end

local function getServers(gameId, pageNumber)
    pageNumber = pageNumber or 1
    local url = string.format("https://games.roblox.com/v1/games/%d/servers/0?sortOrder=1&limit=100&pageNumber=%d", gameId, pageNumber)
    
    local success, response = pcall(function()
        return HttpService:GetAsync(url, true)
    end)
    
    if success then
        local ok, data = pcall(function()
            return HttpService:JSONDecode(response)
        end)
        if ok and data then
            return data.data or {}
        end
    end
    return {}
end

local function scanCurrentServer()
    local brainrots = {}
    local plots = workspace:FindFirstChild("Plots")
    
    if not plots then return brainrots end
    
    for _, folder in pairs(plots:GetDescendants()) do
        if folder:IsA("TextLabel") and folder.Name == "Generation" and folder.Parent and folder.Parent:FindFirstChild("DisplayName") then
            local name = folder.Parent.DisplayName.ContentText
            
            if folder.Parent:FindFirstChild("Mutation") and folder.Parent.Mutation.Visible then
                name = "[" .. folder.Parent.Mutation.ContentText .. "] " .. name
            end
            
            for _, validName in pairs(VALID_BRAINROT_NAMES) do
                if string.lower(name):find(string.lower(validName)) then
                    table.insert(brainrots, {
                        nome = name,
                        valor = folder.ContentText,
                        valorNum = parseMoneyPerSec(folder.ContentText)
                    })
                    break
                end
            end
        end
    end
    
    return brainrots
end

local function searchServers()
    foundServers = {}
    local page = 1
    local totalCount = 0
    
    if statsLabel then
        statsLabel.Text = "üîç BUSCANDO..."
    end
    
    for page = 1, 10 do
        local servers = getServers(placeId, page)
        
        if not servers or #servers == 0 then break end
        
        for _, server in pairs(servers) do
            table.insert(foundServers, {
                id = server.id,
                playerCount = server.playerCount or 0
            })
            totalCount = totalCount + 1
        end
        
        wait(0.3)
    end
    
    table.sort(foundServers, function(a, b)
        return a.playerCount < b.playerCount
    end)
    
    if statsLabel then
        statsLabel.Text = "‚úÖ Encontrados " .. #foundServers .. " servidores"
    end
    
    updateServerList()
    sendWebhook("üîç Busca Completa", "Encontrados: " .. #foundServers .. " servidores", 65280)
end
