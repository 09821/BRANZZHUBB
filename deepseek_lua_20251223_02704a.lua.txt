    -- Efeito de pulso
    spawn(function()

            wait(0.5)
            TweenService:Create(circle, TweenInfo.new(0.5), {Size = UDim2.new(0, 60, 0, 60)}):Play()
            wait(2)
        end
    end)

    -- Sistema de arrastar
    local dragging = false
    local dragStart, startPos

    circle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = circle.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart
            circle.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)

    return circle
end

-- FunÃ§Ã£o: Parsear dinheiro por segundo
local function parseMoneyPerSec(text)
    if not text then return 0 end
    local mult = 1
    local numberStr = text:match("[%d%.]+")
    if not numberStr then return 0 end
    if text:find("K") then mult = 1_000
    elseif text:find("M") then mult = 1_000_000
    elseif text:find("B") then mult = 1_000_000_000
    elseif text:find("T") then mult = 1_000_000_000_000
    elseif text:find("Q") then mult = 1_000_000_000_000_000 end
    return tonumber(numberStr) * mult
end

-- FunÃ§Ã£o: Detectar Brainrots
local function detectBrainrots()
    local found = {}
    
    -- Procurar na pasta Plots
    local plotsFolder = Workspace:FindFirstChild("Plots")
    if plotsFolder then
        for _, plot in ipairs(plotsFolder:GetDescendants()) do
            if plot:IsA("Model") then
                -- Verificar brainrots
                for _, brainrot in ipairs(VALID_BRAINROTS) do
                    if plot.Name == brainrot then
                        table.insert(found, {
                            name = brainrot,
                            object = plot,
                            position = plot:GetPivot().Position
                        })
                        break
                    end
                end
                
                -- Verificar IDs
                for _, id in ipairs(TARGET_IDS_1) do
                    if plot.Name == id then
                        table.insert(found, {
                            name = "ID1: " .. id,
                            object = plot,
                            position = plot:GetPivot().Position
                        })
                    end
                end
                
                for _, id in ipairs(TARGET_IDS_2) do
                    if plot.Name == id then
                        table.insert(found, {
                            name = "ID2: " .. id,
                            object = plot,
                            position = plot:GetPivot().Position
                        })
                    end
                end
            end
        end
    end
    
    -- Procurar por geraÃ§Ã£o de dinheiro
    local valorizado = false
    for _, obj in pairs(Workspace:GetDescendants()) do
        if obj:IsA("TextLabel") and obj.Name == "Generation" then
            local nome = ""
            if obj.Parent:FindFirstChild("Mutation") and obj.Parent.Mutation.Visible then
                nome = "[" .. obj.Parent.Mutation.Text .. "] " .. obj.Parent.DisplayName.Text
            else
                nome = obj.Parent.DisplayName.Text
            end
            
            if parseMoneyPerSec(obj.Text) > 49999999 then
                valorizado = true
                table.insert(found, {
                    name = "ðŸ’Ž " .. nome,
                    object = obj.Parent,
                    position = obj.Parent:GetPivot().Position,
                    value = obj.Text
                })
            end
        end
    end
    
    return found
end
