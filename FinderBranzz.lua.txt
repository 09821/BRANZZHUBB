-- Fun√ß√£o para converter moeda
local function parseMoneyPerSec(text)
    if not text then return 0 end
    local mult = 1
    local numberStr = text:match("[%d%.]+")
    if not numberStr then return 0 end
    if text:find("K") then mult = 1_000
    elseif text:find("M") then mult = 1_000_000
    elseif text:find("B") then mult = 1_000_000_000
    elseif text:find("T") then mult = 1_000_000_000_000
    elseif text:find("Q") then mult = 1_000_000_000_000_000 end
    return tonumber(numberStr) * mult
end

    end
end

-- Procurar brainrots no workspace
local function searchForBrainrots()
    local foundBrainrots = {}
    local valorizadoDog = false
    
    for _, brainrote in pairs(getgenv().base:GetDescendants()) do
        if brainrote:IsA("TextLabel") and brainrote.Name == "Generation" then
            local nomezinDog = ""
            if brainrote.Parent:FindFirstChild("Mutation") and brainrote.Parent.Mutation.Visible then
                nomezinDog = "[" .. brainrote.Parent.Mutation.ContentText .. "] " .. brainrote.Parent.DisplayName.ContentText
            else
                nomezinDog = brainrote.Parent.DisplayName.ContentText
            end
            
            -- Verificar se √© um brainrot da lista
            for _, brainrotName in pairs(BRAINROTS_LIST) do
                if nomezinDog:find(brainrotName) then
                    local value = brainrote.ContentText
                    local parsedValue = parseMoneyPerSec(value)
                    
                    table.insert(foundBrainrots, {
                        name = nomezinDog,
                        value = value,
                        parsedValue = parsedValue,
                        object = brainrote.Parent
                    })
                    
                    if parsedValue > 49999999 then
                        valorizadoDog = true
                    end
                    
                    -- Criar ESP
                    createESP(brainrote.Parent, nomezinDog, value)
                    break
                end
            end
        end
    end
    
    return foundBrainrots, valorizadoDog
end

-- Procurar em Plots
local function searchInPlots()
    local plots = Workspace:FindFirstChild("Plots")
    if not plots then return {} end
    
    local found = {}
    
    for _, plot in pairs(plots:GetChildren()) do
        -- Verificar IDs dos plots
        for _, plotId in pairs(PLOT_IDS) do
            if plot.Name:find(plotId) or plot:GetAttribute("PlotId") == plotId then
                -- Procurar objetos com nomes da lista PICT
                for _, child in pairs(plot:GetDescendants()) do
                    if child:IsA("Model") or child:IsA("Part") then
                        for _, pictName in pairs(PICT_IDS) do
                            if child.Name:find(pictName) then
                                table.insert(found, {
                                    type = "PICT",
                                    name = child.Name,
                                    object = child,
                                    plot = plot.Name
                                })
                                
                                -- Criar ESP
                                createESP(child, "PICT: " .. child.Name, "")
                            end
                        end
                    end
                end
            end
        end
    end
    
    return found
end

-- Buscar servidores
local function findServers()
    if not PLACE_ID then return {} end
    
    local servers = {}
    local success, result = pcall(function()
        local url = string.format("https://games.roblox.com/v1/games/%d/servers/Public?limit=100", PLACE_ID)
        local response = HttpService:GetAsync(url)
        local data = HttpService:JSONDecode(response)
        
        return data.data
    end)
    
    if success and result then
        for _, server in pairs(result) do
            if server.playing > 0 then
                table.insert(servers, {
                    id = server.id,
                    players = server.playing,
                    maxPlayers = server.maxPlayers,
                    ping = server.ping or 0
                })
            end
        end
    end
    
    return servers
end

-- Servidor Hop autom√°tico
local function serverHop()
    if not PLACE_ID or not serverHopActive then return end
    
    while serverHopActive do
        local servers = findServers()
        
        -- Ordenar por menor n√∫mero de players
        table.sort(servers, function(a, b)
            return a.players < b.players
        end)
        
        for _, server in pairs(servers) do
            if server.players <= 1 then
                -- Tentar teleportar
                if teleportToServer(server.id) then
                    -- Aguardar carregamento
                    task.wait(5)
                    
                    -- Procurar brainrots
                    local brainrots, valorizado = searchForBrainrots()
                    local plotsFound = searchInPlots()
                    
                    if #brainrots > 0 or #plotsFound > 0 then
                        -- Enviar webhook
                        local embed = {
                            title = "SERVIDOR ENCONTRADO!",
                            description = string.format("Players in serverüë•: %d/%d\n\nDATE AND TIME ‚è≥: %s\n\nBEST BRAINROTSüß†",
                                server.players, server.maxPlayers, DateTime.now():ToIsoDate()),
                            color = 0x00FF00,
                            fields = {}
                        }
                        
                        -- Adicionar brainrots
                        for i, brainrot in ipairs(brainrots) do
                            if i <= 10 then
                                table.insert(embed.fields, {
                                    name = brainrot.name,
                                    value = brainrot.value,
                                    inline = true
                                })
                            end
                        end
                        
                        -- Adicionar plots encontrados
                        if #plotsFound > 0 then
                            table.insert(embed.fields, {
                                name = "PLOTS ENCONTRADOS",
                                value = #plotsFound .. " plots com objetos especiais",
                                inline = false
                            })
                        end
                        
                        embed.fields[#embed.fields + 1] = {
                            name = "JOIN SERVERüîó",
                            value = string.format("[Clique aqui para entrar](https://www.roblox.com/games/%d?privateServerLinkCode=%s)",
                                PLACE_ID, server.id),
                            inline = false
                        }
                        
                        sendWebhook({
                            content = "@everyone SERVER FOUND!",
                            embeds = {embed}
                        })
                        
                        -- Atualizar GUI
                        if mainGui then
                            updateServerList(server.id, brainrots, plotsFound)
                        end
                        
                        -- Auto TP se ativado
                        if autoTP and valorizado then
                            -- Teleportar para o brainrot mais valioso
                            local bestBrainrot = nil
                            local bestValue = 0
                            
                            for _, brainrot in pairs(brainrots) do
                                if brainrot.parsedValue > bestValue then
                                    bestValue = brainrot.parsedValue
                                    bestBrainrot = brainrot
                                end
                            end
                            
                            if bestBrainrot and bestBrainrot.object then
                                local humanoid = player.Character and player.Character:FindFirstChild("Humanoid")
                                if humanoid then
                                    humanoid:MoveTo(bestBrainrot.object.Position)
                                end
                            end
                        end
                        
                        return true
                    end
                end
            end
        end
        
        task.wait(SERVER_HOP_INTERVAL)
    end
    
    return false
end

-- Iniciar busca de servidores
local function startServerSearch()
    if not PLACE_ID then return end
    
    isSearching = true
    
    task.spawn(function()
        while isSearching do
            local servers = findServers()
            
            if #servers > 0 then
                -- Procurar brainrots no servidor atual
                local brainrots, valorizado = searchForBrainrots()
                local plotsFound = searchInPlots()
                
                if #brainrots > 0 or #plotsFound > 0 then
                    -- Encontraram algo!
                    updateServerList(game.JobId, brainrots, plotsFound)
                end
            end
            
            task.wait(10)
        end
    end)
end

-- Inicializa√ß√£o
local function initialize()
    -- Verificar se j√° est√° em um jogo
    if game.PlaceId > 0 then
        PLACE_ID = game.PlaceId
        createMainGui()
        
        -- Iniciar busca automaticamente
        task.spawn(function()
            task.wait(2)
            startServerSearch()
        end)
    else
        -- Mostrar GUI de input
        createPlaceIdGui()
    end
end

-- Executar quando o script carregar
task.spawn(initialize)

-- Auto-execute quando trocar de servidor
local lastJobId = game.JobId

game:GetService("Players").LocalPlayer.OnTeleport:Connect(function(state)
    if state == Enum.TeleportState.InProgress then
        -- Salvar dados
        local teleportData = {
            PLACE_ID = PLACE_ID,
            autoTP = autoTP,
            serverHopActive = serverHopActive
        }
        
        TeleportService:SetTeleportSetting("BrainrotFinderData", teleportData)
    end
end)

-- Recuperar dados ap√≥s teleporte
task.spawn(function()
    task.wait(1)
    
    local teleportData = TeleportService:GetTeleportSetting("BrainrotFinderData")
    if teleportData then
        PLACE_ID = teleportData.PLACE_ID or PLACE_ID
        autoTP = teleportData.autoTP or false
        serverHopActive = teleportData.serverHopActive or false
        
        if PLACE_ID then
            createMainGui()
            startServerSearch()
            
            if serverHopActive then
                task.spawn(serverHop)
            end
        end
    end
end)
